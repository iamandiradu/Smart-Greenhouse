Here is the function i wrote to display a custom image at a specific X,Y coordinates on the screen :

 
 
 
void imag(const char picture[], char X, char Y, char picture_size, char picture_height)
{
//  X - Y : Upper left corner coordinates of the picture
//  picture_size , picture_height : image width, height in pixels

    char Column,Page = 0;
    int i = 0;

    cursor_OLED(X, Y);        // Set the initial picture position

    for(Page=0; Page<8; Page++)
    {
        cursor_OLED(Column, Y);

        for (Column=0; Column < picture_size; Column++)
           {
           SSD1306_write_Data(picture[i++]);
           }
    }
}

 

 
 
 
void cursor_OLED(unsigned char X,unsigned char Y)
{
 
 
 
#define SET_PAGE            0xB0
 
 
 
#define SET_COL_LO        0x00
 
 
 
#define SET_COL_HI          0x10
 
 
 

SSD1306_write_Cmd( SET_PAGE + X);
SSD1306_write_Cmd( SET_COL_LO + (8*Y & 0x0F));
SSD1306_write_Cmd( SET_COL_HI + ((8*Y >>4) & 0x0F));
}

 
The function is called in the main loop as below :
imag(onair,0,0, 49,64);       // Display 'onair' picture at 0,0 coordinates - Picture size is 49x64 pixels
